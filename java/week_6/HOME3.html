package org.javaro.lecture; public class HOME3 { public static void main(String[] args) { System.out.println("최은서 - 20191010 - 2D 도면상 두 개의 점 거리 구하기"); int x1, y1, x2, y2; x1 = 10; y1 = 30; x2 = 20; y2 = 60; double c = dLength(x1, y1, x2, y2); System.out.println("두 점 사이 거리는 "+c); } static double dLength(int x1, int y1, int x2, int y2) { return Math.sqrt((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1)); } }
