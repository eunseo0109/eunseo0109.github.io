package org.javaro.lecture; public class HOME_4 { public static void main(String[] args) { System.out.println("최은서 - 20191016 - 자바빈 프로그램"); Home4 home = new Home4(); home.setX(10.0); home.setY(20.0); double mult_value = home.fn_mult(); double area1_value = home.fn_area1(); double area2_value = home.fn_area2(); System.out.println("곱하기 : " + mult_value); System.out.println("사각형 면적 : " + area1_value); System.out.println("삼각형 면적 : " + area2_value); } } class Home4{ private double x, y; void setX(double x) { this.x = x; } void setY(double y) { this.y = y; } double getX() { return this.x; } double getY() { return this.y; } public double fn_mult() { return this.x*this.y; } public double fn_area1() { return this.x*this.y; } public double fn_area2() { return 0.5*this.x*this.y; } }
