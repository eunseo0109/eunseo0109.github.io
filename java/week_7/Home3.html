package org.javaro.lecture; public class HOME_3 { public static void main(String[] args) { System.out.println("최은서 - 20191016 - 객체 프로그램"); Home3 home = new Home3(); home.x = 10.0; home.y = 20.0; double mult_value = home.fn_mult(); double area1_value = home.fn_area1(); double area2_value = home.fn_area2(); System.out.println("곱하기 : " + mult_value); System.out.println("사각형 면적 : " + area1_value); System.out.println("삼각형 면적 : " + area2_value); } } class Home3 { double x, y; public double fn_mult() { return x*y; } public double fn_area1() { return x*y; } public double fn_area2() { return 0.5*x*y; } }
