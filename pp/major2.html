- 종업원에 대해 Employee 클래스를 정의하고 Employee 클래스의 멤버 변수들에 값을 세팅하고
불러 올 수 있는 접근자 함수를 정의하고 사용하는 프로그램을 작성한다.
조한 종업원의 월급을 계산하는 합수를 작성하고 출력한다.
#include <iostream> #include <string> using namespace std; class Employee { protected: string m_name; // 시간 int m_payRate; // 시간당 급여 float m_hoursWorked; // 근무한시간 public: Employee(); Employee(string name, int payRate, float hoursWorked); void Set_Name(string name); string Get_Name(); void Set_PayRate(int payRate); int Get_PayRate(); void Set_HoursWorked(float hoursWorked); float Get_HoursWorked(); float Compute_Salary(); void Print_Info(); }; Employee::Employee() { m_name = "NONE"; m_payRate = 0; m_hoursWorked = 0.0; } Employee::Employee(string name, int payRate, float hoursWorked) { m_name = name; m_payRate = payRate; m_hoursWorked = hoursWorked; } void Employee::Set_Name(string name) { m_name = name; } void Employee::Set_PayRate(int payRate) { m_payRate = payRate; } void Employee::Set_HoursWorked(float hoursWorked) { m_hoursWorked = hoursWorked; } string Employee::Get_Name() { return m_name; } int Employee::Get_PayRate() { return m_payRate; } float Employee::Get_HoursWorked() { return m_hoursWorked; } float Employee::Compute_Salary() { return m_payRate * m_hoursWorked; } void Employee::Print_Info() { cout << "이름 : " << m_name << endl; cout << "시간당 급여 : " << m_payRate << "원" << endl; cout << "근무 시간 : " << (int)m_hoursWorked << "시간" << endl; cout << "총급여 : " << (int)Compute_Salary() << "원" << endl << endl; } class Manager : public Employee { bool m_fullTime; //t-> 정규 f->비정규 public: Manager(); Manager(string name, int payRate, float hoursWorked, bool fullTime); bool Get_FullTime(); void set_FullTime(bool fullTime); float Compute_Salary(); void Print_Info(); }; Manager::Manager() { m_fullTime = true; } Manager::Manager(string name, int payRate, float hoursWorked, bool fullTime) : Employee(name, payRate, hoursWorked) { m_fullTime = fullTime; } bool Manager::Get_FullTime() { return m_fullTime; } void Manager::set_FullTime(bool fullTime) { m_fullTime = fullTime; } float Manager::Compute_Salary(){ if (m_fullTime == true) { m_payRate *= 1.5; m_hoursWorked = 40; } return Employee::Compute_Salary(); } void Manager::Print_Info() { if (m_fullTime == true) { cout << "이름 : " << m_name << endl; cout << "시간당 급여 : " << m_payRate << "원" << endl; cout << "정규직 매니저" << endl; cout << "근무시간 40시간 고정" << endl; cout << "총 급여 : " << Manager::Compute_Salary() << "원" << endl << endl; } else { cout << "이름 : " << m_name << endl; cout << "시간당 급여 : " << m_payRate << "원" << endl; cout << "비정규직 매니저" << endl; cout << "근무시간 : " << m_hoursWorked << "시간" << endl; cout << "총 급여 : " << Employee::Compute_Salary() << "원" << endl << endl; } } int main() { Employee m1("장길산", 8000, 32.0); Manager m2("홍길동", 12000, 28.0, true); cout << "<종업원>" << endl; m1.Print_Info(); m1.Set_PayRate(8500); m1.Set_HoursWorked(40.0); cout << "***** 종업원 PayRate 및 hours 변경 *****" << endl; cout << "시간당 급여 : " << m1.Get_PayRate() << "원" << endl; cout << "근무 시간 : " << m1.Get_HoursWorked() << "시간" << endl; cout << "총 급여 : " << m1.Compute_Salary() << "원" << endl << endl; cout << "<매니저1>" << endl; m2.Print_Info(); m2.Set_PayRate(13500); m2.Set_HoursWorked(35.0); cout << "***** 매니저 PayRate 및 hours 변경 *****" << endl; cout << "시간당 급여 : " << m2.Get_PayRate() << "원" << endl; cout << "총 급여 : " << m2.Compute_Salary() << "원" << endl << endl; Manager m3; cout << "<매니저2>" << endl; m3.Print_Info(); m3.Set_Name("박문수"); m3.Set_PayRate(12000); m3.Set_HoursWorked(25.5); m3.set_FullTime(false); cout << "***** 매니저 이름, PayRate 및 hours 변경 *****" << endl; m3.Print_Info(); return 0; }
